//table[@class = 'dataTable']/thead/tr - represents the header row
//table[@class = 'dataTable']/thead/tr/th - represents each col of the header row
//table[@class = 'dataTable']/thead/tr/th[1] - represents first col of the header row
//table[@class = 'dataTable']/thead/tr/th[2] - represents second col of the header row

//table[@class = 'dataTable']/tbody - represents the webtable without the header row
//table[@class = 'dataTable']/tbody/tr - represents total number of rows in the webtable
//table[@class = 'dataTable']/tbody/tr[9] - represents 9th row of the webtable
//table[@class = 'dataTable']/tbody/tr[9]/td - represents all cols in the 9th row
//table[@class = 'dataTable']/tbody/tr[9]/td[3] - represents 3rd col of 9th row
//table[@class = 'dataTable']/tbody/tr/td[1] - represents all values in the 1st col

//table[@class = 'dataTable']/tbody/tr/td[2] - represents all values in the 2nd col
